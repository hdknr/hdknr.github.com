DocumentRoot does not exist ? あるのに¶

SELinux! :

# getenforce Enforcing

無効:

# setenforce 0 # getenforce Permissive

リブート時に無効にさせる:

# vi /etc/sysconfig/selinux SELINUX=disabled

macro.httpd.xml¶

<Macro VHost $host $base $port>
<VirtualHost *:$port>
#Basic
	ServerAdmin admin@$host
	ServerName  $host
	DocumentRoot $base/$host/www/

#Proxy
    ProxyRequests off

# BasicAuth
   <Location "/" >
	Use Auth $base $host user
   </Location>

# Virtual Directories and Others
    Include $base/$host/conf/vdir.*.conf
 
#Log
	ErrorLog $base/$host/logs/error.log
	LogLevel warn
	CustomLog $base/$host/logs/access.log combined

# SSL
#	Include $base/$host/conf/httpd.conf.d/ssl.def
</VirtualHost>
</Macro>

httpd.conf¶

  • macro.httpd.xml をつかって httpd.conf を定義する。

  • 以下の例は dev.lafoglia.jp サイトの仮想サーバーの例。

    # mod_macro skelton
Include /home/hdknr/sites/dev.lafoglia.jp/conf/macro.auth.xml
Include /home/hdknr/sites/dev.lafoglia.jp/conf/macro.httpd.xml
Include /home/hdknr/sites/dev.lafoglia.jp/conf/macro.jenkins.xml
#
Use VHost dev.lafoglia.jp /home/hdknr/sites  80
#

  • これをシンボリックリンクする

    $ ls -l /etc/apache2/sites-enabled/

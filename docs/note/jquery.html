フォーム送信の結果で部分画面を入れ替える¶

function blockpost(form,container)
{
    $.ajaxSettings.traditional = true; // For Django, not for PHP
    $.post(
        $(form).attr('action'),
        $(form).serialize(),
        function(data) {
            $(form).parents(container).html(data);
        }
    );
}

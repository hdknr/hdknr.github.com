preg_match¶

  • URLのマッチング
$pattern='/^(?:(?P<scheme>https*))?'.
         '(?:[:\/]+)?'.
         '(?P<host>[^\?#\/:]+)'.
         '(?::(?P<port>\d+))?'.
         '(?P<path>[^#\?]*)'.
         '(?:\?(?P<query>[^#]+))?'.
         '(?:#(?P<fragment>.*))?/';

$matches = Array();
$url = "http://hoge.com:9000/path#frag=xxx";
if( preg_match( $pattern, $url ,$matches)) {
    echo( var_dump( $matches ) );
}
